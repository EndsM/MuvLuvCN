on_actions = {
	# initiate variables
	on_startup = {
		effect = {
			
		}
	}
	# log the casualties
	on_capitulation_immediate = {
		effect = {
			# 与难民解放战线结束战争的情况
			if = {
				limit = {
					ROOT = {
						tag = M01
					}
				}
			}
			# For BETA
			if = {
				limit = {
					ROOT = {
						OR = {
							is_in_faction_with = BET
							original_tag = BET
							original_tag = H00
						}
					}
				}
				global = {
					# add casualties to global var
					set_variable = {
						var = BETA_casualties
						value = ROOT.casualties
					}
				}
			}
			# For Human
		}
	}
	on_capitulation = {
		effect = {
		}
	}
	# remove flags for log again
	on_uncapitulation = {
		effect = {
		}
	}
}
